 17        """
 18        Given distribution parameters generate a random measurement result.
 19        
 20        Args:
 21            kappa (int): The kappa distribution parameter.
 22            eta (int): The eta distribution parameter.

 27

 29            >>> _generate_measurement_outcome(2, 3, 5)
 30            {2, 0, 3, 1, 4}
 31            >>> _generate_measurement_outcome(0, 2, 4)
 32            {0, 2, 0, 2}

 41            return (kappa + distribution) % dimension

111        """
112        Add the generators at column index2 to the generators at column index1.

179    def _eliminate_columns(self, qudit_index: int, row: np.ndarray) -> np.ndarray:
180        """
181        Eliminate columns in the row.
182
183        Args:
184            qudit_index (int): Index of the qudit.
185            row (np.ndarray): Row to process.
186
187        Returns:
188            np.ndarray: Processed row.
189
190        Raises:
191            ValueError: If a suitable column to swap cannot be found.
192        """
193        row_gcd = np.gcd.reduce(row)
194        cols = row.shape[0]
195        gcd_col = False
196        coprime_col = False
197        pivot_col = None
198        for col in range(cols):
199            if row[col] in self.coprime_order:
200                coprime_col = True
201                pivot_col = col
202                break
203            if row[col] == row_gcd:
204                gcd_col = True
205                pivot_col = col
206                break
207        if not coprime_col and not gcd_col:
208            # Find a pair of columns with gcd equal to row_gcd
209            for i in range(cols):
210                for j in range(i + 1, cols):
211                    if np.gcd(row[i], row[j]) == row_gcd:
212                        x, y, _ = self._extended_euclidean(row[i], row[j])
213                        x, y = x % self.order, y % self.order
214                        if x in self.coprime_order:
215                            self.multiply_generator(i, x)
216                            self.add_generators(i, j, y)
217                            pivot_col = i
218                            gcd_col = True
219                        elif y in self.coprime_order:
220                            self.multiply_generator(j, y)
221                            self.add_generators(j, i, x)
222                            pivot_col = j
223                            gcd_col = True
224                        else:
225                            raise ValueError("Could not find suitable column to swap.")
226                        break
227                if gcd_col:
228                    break
229        if coprime_col:
230            # Calculate the multiplicative inverse of the pivot element modulo self.order
231            pivot = int(row[pivot_col])
232            inv_pivot = pow(pivot, -1, self.order)
233            # Eliminate other columns using the pivot column
234            for i in range(cols):
235                if i != pivot_col and row[i] != 0:
236                    factor = (-int(row[i]) * inv_pivot) % self.order
237                    self.add_generators(i, pivot_col, factor)
238        if gcd_col:
239            # eliminate other columns using the gcd_col
240            pivot = int(row[pivot_col])
241            for i in range(cols):
242                target = int(row[i])
243                if i != pivot_col and target != 0:
244                    g = gcd(pivot, self.order)
245                    if row[i] % g == 0:
246                        factor = ((-target // g) * pow(pivot // g, -1, self.order // g)) % (self.order // g)
247                        self.add_generators(i, pivot_col, factor)
248        last_col = cols - 1
249        self.swap_generators(pivot_col, last_col)
250        return row

252    def _get_eta_as_divisor(self, qudit_index: int, row: np.ndarray):
253        """
254        Find suitable alpha such that eta as a divisor of the dimension.
255        Will attempt to find add a multiple of the dimension (multiply by identity) if no suitable alpha is found.
256
257        Args:
258            qudit_index (int): Index of the qudit.
259            row (np.ndarray): Row to process.
260
261        Returns:
262            int or None: The eta value or None if not found.
263        """
264        last_col = row.shape[0] - 1
265        if row[-1] != 0:
266            if self.dimension % row[-1]== 0:
267                return row[-1]
268            else:
269                for alpha in self.coprime_order:
270                    value = (row[-1] * alpha) % self.order
271                    if self.dimension % value == 0:
272                        self.multiply_generator(last_col, alpha)
273                        return value
274            # reduce the row modulo the dimension
275            if row[-1] > self.dimension:
276                self.x_block[qudit_index, -1] -= self.dimension
277                self.phase_vector[-1] -= self.dimension//2
278            else:
279                self.x_block[qudit_index, -1] += self.dimension
280                self.phase_vector[-1] += self.dimension//2
281            for alpha in self.coprime_order:
282                    value = (row[-1] * alpha) % self.order
283                    if self.dimension % value == 0:
284                        self.multiply_generator(last_col, alpha)
285                        return value
286        return None

314    def _handle_non_deterministic_case(self, new_stabilizer: np.ndarray, s: int, qudit_index: int, measurement_value: int) -> MeasurementResult:
315        """
316        Handle the non-deterministic case of measurement.
317
318        Args:
319            new_stabilizer (np.ndarray): New stabilizer to add.
320            s (int): The s value.
321            qudit_index (int): Index of the qudit.
322            measurement_value (int): The measurement value.
323
324        Returns:
325            MeasurementResult: The result of the measurement.
326        """
327        last_column = self.stab_tableau[:, -1] * s
328        last_column_index = self.stab_tableau.shape[1] - 1
329
330        if self._is_last_column_identity(last_column):
331            self.update(new_stabilizer, last_column_index)
332        else:
333            excluding_commuting = self._prepare_excluding_commuting_matrix(new_stabilizer)
334            try:
335                result = solve(excluding_commuting, last_column)
336            except Exception as e:
337                try:
338                    result = dp.solve(Matrix(excluding_commuting), Matrix(last_column))
339                except Exception as e:
340                    result = True 
341            if not result:
342                self.multiply_generator(last_column_index, s, allow_non_coprime=True)
343                self.append(new_stabilizer)
344            else:
345                self.update(new_stabilizer, last_column_index)
346
347        return MeasurementResult(qudit_index=qudit_index, deterministic=False, measurement_value=measurement_value)

367    def _extended_euclidean(self, a: int, b: int) -> Tuple[int, int, int]:
368        """
369        Compute the extended Euclidean algorithm for a and b.
370
371        Args:
372            a (int): First number.
373            b (int): Second number.
374
375        Returns:
376            Tuple[int, int, int]: x, y, and gcd(a, b) such that a*x + b*y = gcd(a, b).
377
378        Raises:
379            ValueError: If suitable x and y cannot be found.
380        """
381        x, y, u, v = 1, 0, 0, 1
382        while b != 0:
383            q, r = a // b, a % b
384            m, n = x - u * q, y - v * q
385            a, b, x, y, u, v = b, r, u, v, m, n
386
387        # Ensure x and y are positive
388        if a < 0:
389            x, y = -x, -y
390
391        # Check if one of x or y is coprime to self.order
392        if x % self.order in self.coprime_order or y % self.order in self.coprime_order:
393            return x, y, a
394        raise ValueError("Could not find suitable x and y.")

407    def column_reduction(self, tableau_matrix: np.ndarray, weyl_vector: np.ndarray, s: int) -> Optional[int]:
408        """
409        Perform column reduction on the tableau matrix.
410
411        Args:
412            tableau_matrix (np.ndarray): The tableau matrix.
413            weyl_vector (np.ndarray): The Weyl vector.
414            s (int): The s value.
415
416        Returns:
417            Optional[int]: The result of the column reduction, or None if not found.
418        """
419        pauli_vector = np.hstack((0, weyl_vector)).reshape(-1, 1) 
420        full_tableau = np.hstack((tableau_matrix, -s*pauli_vector)) % self.order
421        rows = full_tableau.shape[0] 
422        cols = full_tableau.shape[1]
423        pivot_row = 1
424        for col in range(cols):
425            if pivot_row >= rows:
426                break
427            coprime = False
428            gcd_col = False
429            pivot_col = col
430            for row in range(pivot_row, rows):
431                if np.all(full_tableau[row, col:] == 0): # if everything to the right including pivot is zero, check next row
432                    continue
433                elif np.all(full_tableau[row, col+1:] == 0): # if everything to the right of the pivot is zero, go to next column if everything below is zero
434                    if row < rows - 1 and np.all(full_tableau[row+1:, col] == 0):
435                        break
436                    continue
437                if full_tableau[row, -1] != 0: # if the last column is non-zero, check if the element above is non zero and that everything between pivot and last column is non zero 
438                    if self.num_qudits > 1 and row > 1:
439                        if np.any(full_tableau[row, col+1:-1]) and full_tableau[row-1, col] != 0:
440                            break
441                row_gcd = np.gcd.reduce(full_tableau[row, col:])
442                for i in range(col, cols-1):
443                    if full_tableau[row, i] in self.coprime_order:
444                        pivot_row = row
445                        coprime = True
446                        if pivot_col != i:
447                            self._swap_columns_matrix(full_tableau, col, i)
448                        break
449                    if full_tableau[row, i] == row_gcd:
450                        pivot_row = row
451                        gcd_col = True
452                        if pivot_col != i:
453                            self._swap_columns_matrix(full_tableau, col, i)
454                        break
455                if not coprime and not gcd_col:
456                    for i in range(col, cols-1):
457                        for j in range(i + 1, cols-1):
458                            if np.gcd(full_tableau[row, i], full_tableau[row, j]) == row_gcd:
459                                # Solve Bezout's identity to get the column with row_gcd
460                                x, y, _ = self._extended_euclidean(full_tableau[row, i], full_tableau[row, j])
461                                x, y = x % self.order, y % self.order
462                                if x in self.coprime_order:
463                                    full_tableau[:, i] *= x
464                                    full_tableau[:, i] %= self.order
465                                    self._add_column_matrix(full_tableau, j, i, y)
466                                    
467                                elif y in self.coprime_order:
468                                    full_tableau[:, j] *= y
469                                    full_tableau[:, j] %= self.order
470                                    self._add_column_matrix(full_tableau, i, j, x)
471                                    self._swap_columns_matrix(full_tableau, i, j)
472                                else:
473                                    raise ValueError("Could not find suitable column to swap.")
474                                if pivot_col != i:
475                                    self._swap_columns_matrix(full_tableau, col, i)
476                                pivot_row = row
477                                gcd_col = True
478                                break
479                        if coprime or gcd_col:
480                            break
481                if coprime or gcd_col:
482                    break
483            if coprime:
484                # Calculate the multiplicative inverse of the pivot element modulo self.order
485                pivot = int(full_tableau[pivot_row, pivot_col])
486                inv_pivot = pow(pivot, -1, self.order)
487                # Eliminate other columns using the pivot column
488                for i in range(pivot_col, cols):
489                    if i != pivot_col and full_tableau[pivot_row, i] != 0:
490                        factor = (-int(full_tableau[pivot_row, i]) * inv_pivot) % self.order
491                        self._add_column_matrix(full_tableau, pivot_col, i, factor)
492            if gcd_col:
493                # eliminate other columns using the gcd_col
494                pivot = int(full_tableau[pivot_row, pivot_col])
495                for i in range(pivot_col, cols):
496                    target = int(full_tableau[pivot_row, i])
497                    if i != pivot_col and target != 0:
498                        g = gcd(pivot, self.order)
499                        if full_tableau[pivot_row, i] % g == 0:
500                            factor = ((-target // g) * pow(pivot // g, -1, self.order // g)) % (self.order // g)
501                            self._add_column_matrix(full_tableau, pivot_col, i, factor)
502            pivot_row += 1
503        return full_tableau[0, -1]

505    def _create_measurement_result(self, t: int, eta: int, s: int, qudit_index: int, weyl_vector: np.ndarray) -> MeasurementResult:
506        """

528    def t_diophantine(self, tableau_matrix: np.ndarray, weyl_vector: np.ndarray, qudit_index: int, s: int) -> Optional[int]:
529        """
530        Solve whether a tableau contains the weyl vector with a given phase value t is in the column span by solving Ax=b.
531        The modulo constraints can be represented as a linear system of Diophantine equations.
532
533        Args:
534            tableau_matrix (np.ndarray): The tableau matrix.
535            weyl_vector (np.ndarray): The Weyl vector.
536            qudit_index (int): Index of the qudit.
537            s (int): The s value.
538
539        Returns:
540            Optional[int]: The solution t, or None if not found.
541        """
542        modulo_constraint = np.ones((self.pauli_size, 1), dtype=np.int64) * self.order
543        tableau_matrix = np.hstack((tableau_matrix, modulo_constraint))
544        if self.even:
545            tableau_matrix[0, qudit_index+self.num_qudits] = (-self.dimension * s // 2) % self.order
546            identity = np.zeros((self.pauli_size, 1), dtype=np.int64)
547            identity[0] = self.dimension
548            tableau_matrix = np.hstack((identity, tableau_matrix))
549        for t in range(self.dimension):
550            solution = np.hstack((s*t, s*weyl_vector)) % self.order
551            try:
552                if solve(tableau_matrix, solution):
553                    return t

161    def _get_single_eta(self, qudit_index: int):
162        """
163        Get the eta value assuming a Z measurement at specified index.
164
165        Args:
166            qudit_index (int): Index of the qudit.
167
168        Returns:
169            int: The eta value.
170        """
171        row = self.x_block[qudit_index, :]
172        
173        if np.all(row == 0):
174            return self.dimension
175        
176        if self.x_block.shape[1] > 1:
177            row = self._eliminate_columns(qudit_index, row)
178        return self._get_eta_as_divisor(qudit_index, row)

180    def _eliminate_columns(self, qudit_index: int, row: np.ndarray) -> np.ndarray:
181        """
182        Eliminate columns in the row.
183
184        Args:
185            qudit_index (int): Index of the qudit.
186            row (np.ndarray): Row to process.
187
188        Returns:
189            np.ndarray: Processed row.
190
191        Raises:
192            ValueError: If a suitable column to swap cannot be found.
193        """
194        row_gcd = np.gcd.reduce(row)
195        cols = row.shape[0]
196        gcd_col = False
197        coprime_col = False
198        pivot_col = None
199        for col in range(cols):
200            if row[col] in self.coprime_order:
201                coprime_col = True
202                pivot_col = col
203                break
204            if row[col] == row_gcd:
205                gcd_col = True
206                pivot_col = col
207                break
208        if not coprime_col and not gcd_col:
209            # Find a pair of columns with gcd equal to row_gcd
210            for i in range(cols):
211                for j in range(i + 1, cols):
212                    if np.gcd(row[i], row[j]) == row_gcd:
213                        x, y, _ = self._extended_euclidean(row[i], row[j])
214                        x, y = x % self.order, y % self.order
215                        if x in self.coprime_order:
216                            self.multiply_generator(i, x)
217                            self.add_generators(i, j, y)
218                            pivot_col = i
219                            gcd_col = True
220                        elif y in self.coprime_order:
221                            self.multiply_generator(j, y)
222                            self.add_generators(j, i, x)
223                            pivot_col = j
224                            gcd_col = True
225                        else:
226                            raise ValueError("Could not find suitable column to swap.")
227                        break
228                if gcd_col:
229                    break
230        if coprime_col:
231            # Calculate the multiplicative inverse of the pivot element modulo self.order
232            pivot = int(row[pivot_col])
233            inv_pivot = pow(pivot, -1, self.order)
234            # Eliminate other columns using the pivot column
235            for i in range(cols):
236                if i != pivot_col and row[i] != 0:
237                    factor = (-int(row[i]) * inv_pivot) % self.order
238                    self.add_generators(i, pivot_col, factor)
239        if gcd_col:
240            # eliminate other columns using the gcd_col
241            pivot = int(row[pivot_col])
242            for i in range(cols):
243                target = int(row[i])
244                if i != pivot_col and target != 0:
245                    g = gcd(pivot, self.order)
246                    if row[i] % g == 0:
247                        factor = ((-target // g) * pow(pivot // g, -1, self.order // g)) % (self.order // g)
248                        self.add_generators(i, pivot_col, factor)
249        last_col = cols - 1
250        self.swap_generators(pivot_col, last_col)
251        return row

253    def _get_eta_as_divisor(self, qudit_index: int, row: np.ndarray):
254        """
255        Find suitable alpha such that eta as a divisor of the dimension.
256        Will attempt to find add a multiple of the dimension (multiply by identity) if no suitable alpha is found.
257
258        Args:
259            qudit_index (int): Index of the qudit.
260            row (np.ndarray): Row to process.
261
262        Returns:
263            int or None: The eta value or None if not found.
264        """
265        last_col = row.shape[0] - 1
266        if row[-1] != 0:
267            if self.dimension % row[-1]== 0:
268                return row[-1]
269            else:
270                for alpha in self.coprime_order:
271                    value = (row[-1] * alpha) % self.order
272                    if self.dimension % value == 0:
273                        self.multiply_generator(last_col, alpha)
274                        return value
275            # reduce the row modulo the dimension
276            if row[-1] > self.dimension:
277                self.x_block[qudit_index, -1] -= self.dimension
278                self.phase_vector[-1] -= self.dimension//2
279            else:
280                self.x_block[qudit_index, -1] += self.dimension
281                self.phase_vector[-1] += self.dimension//2
282            for alpha in self.coprime_order:
283                    value = (row[-1] * alpha) % self.order
284                    if self.dimension % value == 0:
285                        self.multiply_generator(last_col, alpha)
286                        return value
287        return None

315    def _handle_non_deterministic_case(self, new_stabilizer: np.ndarray, s: int, qudit_index: int, measurement_value: int) -> MeasurementResult:
316        """
317        Handle the non-deterministic case of measurement.
318
319        Args:
320            new_stabilizer (np.ndarray): New stabilizer to add.
321            s (int): The s value.

331        if self._is_last_column_identity(last_column):

368    def _extended_euclidean(self, a: int, b: int) -> Tuple[int, int, int]:
369        """
370        Compute the extended Euclidean algorithm for a and b.
371
372        Args:
373            a (int): First number.
374            b (int): Second number.
375
376        Returns:
377            Tuple[int, int, int]: x, y, and gcd(a, b) such that a*x + b*y = gcd(a, b).
378
379        Raises:
380            ValueError: If suitable x and y cannot be found.
381        """
382        x, y, u, v = 1, 0, 0, 1
383        while b != 0:
384            q, r = a // b, a % b
385            m, n = x - u * q, y - v * q
386            a, b, x, y, u, v = b, r, u, v, m, n
387
388        # Ensure x and y are positive
389        if a < 0:
390            x, y = -x, -y
391
392        # Check if one of x or y is coprime to self.order
393        if x % self.order in self.coprime_order or y % self.order in self.coprime_order:
394            return x, y, a
395        raise ValueError("Could not find suitable x and y.")

413            tableau_matrix (np.ndarray): The tableau matrix.

419        """
420        pauli_vector = np.hstack((0, weyl_vector)).reshape(-1, 1)

422        rows = full_tableau.shape[0]

429            gcd_col = False
430            pivot_col = col
431            for row in range(pivot_row, rows):
432                if np.all(full_tableau[row, col:] == 0): # if everything to the right including pivot is zero, check next row
433                    continue
434                elif np.all(full_tableau[row, col+1:] == 0): # if everything to the right of the pivot is zero, go to next column if everything below is zero
435                    if row < rows - 1 and np.all(full_tableau[row+1:, col] == 0):

437                    continue
438                if full_tableau[row, -1] != 0: # if the last column is non-zero, check if the element above is non zero and that everything between pivot and last column is non zero

443                for i in range(col, cols-1):

445                        pivot_row = row
446                        coprime = True
447                        if pivot_col != i:
448                            self._swap_columns_matrix(full_tableau, col, i)

451                        pivot_row = row
452                        gcd_col = True
453                        if pivot_col != i:

461                                x, y, _ = self._extended_euclidean(full_tableau[row, i], full_tableau[row, j])

463                                if x in self.coprime_order:

477                                pivot_row = row
478                                gcd_col = True

486                pivot = int(full_tableau[pivot_row, pivot_col])

488                # Eliminate other columns using the pivot column
489                for i in range(pivot_col, cols):

491                        factor = (-int(full_tableau[pivot_row, i]) * inv_pivot) % self.order
492                        self._add_column_matrix(full_tableau, pivot_col, i, factor)

495                pivot = int(full_tableau[pivot_row, pivot_col])

500                        if full_tableau[pivot_row, i] % g == 0:

502                            self._add_column_matrix(full_tableau, pivot_col, i, factor)
